class Solution {
    public int[] runningSum(int[] nums) {
        int sum = 0;
        int n = nums.length;
        int ans[] = new int[n];
        for(int i = 1;i<nums.length;i++){
                ans[i] = ans[i-1] + nums[i];
          
        }
      
        return ans;
        
    }
        ans[0] = nums[0];
}
